def Filter():
    try:
        Winedata['Price'] = Winedata['Price'].str.replace('$', '').astype(float)   
    except AttributeError:
        pass
    try:
        Winedata['Alcohol/Vol:'] = Winedata['Alcohol/Vol:'].str.replace('%', '').astype(float)
    except AttributeError:
        pass
    try:
       
        Winedata['Bottle Size:'] = Winedata['Bottle Size:'].str.replace('mL bottle', '')
        Winedata['Bottle Size:'] = Winedata['Bottle Size:'].str.replace('mL tetr', '')
        Winedata['Bottle Size:'] = Winedata['Bottle Size:'].str.replace('mL c', '')
        Winedata['Bottle Size:'] = Winedata['Bottle Size:'].str.replace(' ','')
        Winedata['Bottle Size:'] = Winedata['Bottle Size:'].replace('',np.nan)
        
    except ValueError:
        Winedata['Bottle Size:'] = Winedata['Bottle Size:'].replace(Winedata['Bottle Size:'].unique()[-1], np.nan)
        print('Bottle size error  Possibly: '+Winedata['Bottle Size:'].unique()[-1])# suggests value for .replace(###) to fix error
        pass
        # Winedata['Bottle Size:'] = Winedata['Bottle Size:'].str.replace('mL bottle', '')replace('mL tetr', '').astype(float)
    except AttributeError:
        pass
    try:
        Winedata['Sugar Content:'] = Winedata['Sugar Content:'].str.replace('g/L', '').astype(float)
    except AttributeError:
        pass
    try:
        Winedata['Case Price'] = Winedata['Case Price'].str.replace('/CASE', '')
        Winedata['Case Price'] = Winedata['Case Price'].str.replace('$', '')
        Winedata['Case Price'] = Winedata['Case Price'].replace(' ', np.nan)
        Winedata['Case Price'] = Winedata['Case Price'].str.replace('CASE', '')
        #Winedata['Case Price']= Winedata['Case Price'].astype(float)

    except AttributeError:
        pass
    try:
        Winedata.drop('Unnamed: 0',axis=1,inplace=True)
        Winedata.drop('This is a Kosher product.',axis=1,inplace=True) #Kosher and VQA was not recorded properly for first 3000 wines setall NA's
        Winedata.drop('This is a VQA wine',axis=1,inplace=True)
    except KeyError:
        pass
    print('Wine is Filtered and stored as Winedata')

def CosLEHOT(wineID):#Combined Label enc and One Hot encoding 
    LEHOT=pd.merge(CosineOneHotEnc(wineID), CosineLabelEnc(wineID), how='left',left_index=True,right_index=True)
    SimilarList=pd.merge(LEHOT,FilteredWine, how='left',left_index=True,right_index=True)
    return SimilarList
